# list the todos tied up with it's user_id

todos = {
  1:{
    'text':'todos1', 
    'user_id': 1
  },
  2:{
    'text': 'todos2',
    'user_id':1
  },
  3:{
    'text': 'todos3',
    'user_id': 2
  },
  4:{
    'text': 'todos4',
    'user_id': 1
  },
  5:{
    'text': 'todos5',
    'user_id': 2
  }
}

user_id = 2
user_todos = []
for user_todo in todos.values():
  if user_todo.get('user_id') == user_id:
    user_todos.append(user_todo)
    print(user_todos)
    print(len(user_todos))
